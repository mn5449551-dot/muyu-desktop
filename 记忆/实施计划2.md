│ // 生成主动打招呼消息（非流式，不经过 sendMessage 流）                                                                       │
│ async generateProactiveGreeting({ sessionId, proactiveType, daysSince, chatSystemPrompt }) {                                 │
│   const credentials = this.db.getLlmCredentials()                                                                            │
│   if (!credentials.apiKey) return null                                                                                       │
│                                                                                                                              │
│   // 复用 buildRuntimeContext 获取标准 system messages                                                                       │
│   const contextMessages = this.buildRuntimeContext(sessionId, 5, chatSystemPrompt)                                           │
│                                                                                                                              │
│   // 注入主动触发原因（追加在所有 system 消息之后、user 消息之前）                                                           │
│   const reason = proactiveType === 'birthday'                                                                                │
│     ? '今天是用户的生日，请以真诚温暖的方式向用户送上生日祝福。'                                                             │
│     : `用户已 ${daysSince} 天没有打开应用，请主动问候，表达想念或关心，语气自然不刻意。`                                     │
│   contextMessages.push({ role: 'system', content: reason })                                                                  │
│                                                                                                                              │
│   // 触发用消息（不存 DB，仅用于满足 API 格式要求）                                                                          │
│   contextMessages.push({ role: 'user', content: '（主动问候触发，无用户输入）' })                                            │
│                                                                                                                              │
│   const result = await this.requestNonStreamCompletion({                                                                     │
│     credentials,                                                                                                             │
│     temperature: 0.8,                                                                                                        │
│     messages: contextMessages,                                                                                               │
│   })                                                                                                                         │
│   return String(result || '').trim() || null                                                                                 │
│ }                                                                                                                            │

│ const { daysDiff, toLocalMMDD } = require('./utils/time')                                                                    │
│                                                                                                                              │
│ async function checkProactiveGreeting() {                                                                                    │
│   if (!db.getBooleanAppStateValue('proactive_enabled', false)) return                                                        │
│                                                                                                                              │
│   const now = Date.now()                                                                                                     │
│   const todayUTC = new Date(now).toISOString().slice(0, 10)                                                                  │
│                                                                                                                              │
│   // ---- 缺席触发 ----                                                                                                      │
│   const lastOpenAt = parseInt(db.getAppStateValue('last_open_at') || '0', 10)                                                │
│   const absentDays = lastOpenAt ? daysDiff(now, lastOpenAt) : 0                                                              │
│   const absenceLockKey = `proactive_absence_${todayUTC}`                                                                     │
│                                                                                                                              │
│   let proactiveType = null, daysSince = 0                                                                                    │
│                                                                                                                              │
│   if (absentDays > 3 && !db.getAppStateValue(absenceLockKey)) {                                                              │
│     db.setAppStateValue(absenceLockKey, '1')                                                                                 │
│     proactiveType = 'absence'                                                                                                │
│     daysSince = absentDays                                                                                                   │
│   }                                                                                                                          │
│                                                                                                                              │
│   // ---- 生日触发（缺席与生日不叠加，缺席优先）----                                                                         │
│   if (!proactiveType) {                                                                                                      │
│     const birthday = db.getUserProfileField('birthday')  // 'MM-DD' or null                                                  │
│     if (birthday) {                                                                                                          │
│       const todayMMDD = toLocalMMDD(now)                                                                                     │
│       const birthdayLockKey = `proactive_birthday_${todayUTC}`                                                               │
│       if (todayMMDD === birthday && !db.getAppStateValue(birthdayLockKey)) {                                                 │
│         db.setAppStateValue(birthdayLockKey, '1')                                                                            │
│         proactiveType = 'birthday'                                                                                           │
│       }                                                                                                                      │
│     }                                                                                                                        │
│   }                                                                                                                          │
│                                                                                                                              │
│   // 更新 last_open_at（无论是否触发主动发话都要更新）                                                                       │
│   db.setAppStateValue('last_open_at', String(now))                                                                           │
│                                                                                                                              │
│   if (!proactiveType) return                                                                                                 │
│                                                                                                                              │
│   // 获取当前角色信息                                                                                                        │
│   const charId = getCurrentCharId()                                                                                          │
│   const sessionId = `pet_${charId}`                                                                                          │
│   const char = db.getCharacterById(charId)                                                                                   │
│   const chatSystemPrompt = db.getChatSystemPrompt(charId)                                                                    │
│                                                                                                                              │
│   try {                                                                                                                      │
│     // 非流式生成打招呼消息                                                                                                  │
│     const greeting = await llmService.generateProactiveGreeting({                                                            │
│       sessionId,                                                                                                             │
│       proactiveType,                                                                                                         │
│       daysSince,                                                                                                             │
│       chatSystemPrompt,                                                                                                      │
│     })                                                                                                                       │
│     if (!greeting) return                                                                                                    │
│                                                                                                                              │
│     // 存为 assistant 消息                                                                                                   │
│     db.addChatMessage('assistant', greeting, sessionId)                                                                      │
│                                                                                                                              │
│     // 打开聊天窗口                                                                                                          │
│     openChatWindow({ charId })                                                                                               │
│                                                                                                                              │
│     // 通知 ChatWindowView 刷新消息列表（延迟确保窗口已加载）                                                                │
│     setTimeout(() => {                                                                                                       │
│       sendToAllWindows(IPC.CHAT_RELOAD, { sessionId })                                                                       │
│     }, 500)                                                                                                                  │
│   } catch (err) {                                                                                                            │
│     log.warn('[proactive-greeting]', err.message)                                                                            │
│     // 失败时仍打开聊天窗口，只是没有主动消息                                                                                │
│     openChatWindow({ charId })                                                                                               │
│   }                                                                                                                          │
│ }                                                                                                                            │
