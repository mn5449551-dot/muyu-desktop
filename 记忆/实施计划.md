### P0-1：建立迁移框架

**目标：** 引入版本化迁移，替代现有散落的 ALTER TABLE，为后续加表/加字段提供基础。

**改动文件：** `src/main/db.js`

**实现：**

在 `initializeDB()` 末尾（现有所有 ALTER TABLE 块之后）新增 `runMigrations(db)` 并调用：

```javascript
function runMigrations() {
  const current = parseInt(getAppStateValue('schema_version') || '0', 10)

  const migrations = [
    // v1: 新增 summary_history 表（摘要存档 + token 估算记录）
    () => db.exec(`
      CREATE TABLE IF NOT EXISTS summary_history (
        id            INTEGER PRIMARY KEY AUTOINCREMENT,
        session_id    TEXT NOT NULL,
        summary_text  TEXT NOT NULL,
        from_msg_id   INTEGER NOT NULL,
        to_msg_id     INTEGER NOT NULL,
        input_tokens  INTEGER NOT NULL DEFAULT 0,
        output_tokens INTEGER NOT NULL DEFAULT 0,
        created_at    INTEGER NOT NULL
      )
    `),

    // v2: 新增 character_memory 表（角色关系状态）
    () => db.exec(`
      CREATE TABLE IF NOT EXISTS character_memory (
        session_id         TEXT PRIMARY KEY,
        relationship_stage TEXT NOT NULL DEFAULT 'new',
        summary_count      INTEGER NOT NULL DEFAULT 0,
        updated_at         INTEGER NOT NULL DEFAULT 0
      )
    `),

    // v3: memory_summaries 加软删除字段
    () => {
      try {
        db.exec(`ALTER TABLE memory_summaries ADD COLUMN deleted_at INTEGER DEFAULT NULL`)
      } catch (e) {
        if (!e.message.includes('duplicate column')) throw e
      }
    },
  ]

  for (let i = current; i < migrations.length; i++) {
    migrations[i]()
    setAppStateValue('schema_version', String(i + 1))
  }
}
```

在 `initializeDB()` 末尾调用：`runMigrations()`

**回归验证：**
- 全新安装：`schema_version` = `'3'`，三张新表存在
- 已有数据重启：跳过已执行的 migration，无报错
- 重复启动：幂等，无副作用

### P0-3：时间语义统一

**目标：** 所有时间戳用 UTC 毫秒（`Date.now()`），缺席/生日判断不依赖本地时区。

**改动文件：** `src/main/utils/time.js`（新建）

```javascript
'use strict'

// UTC 天数差（用于"缺席 N 天"判断，不依赖本地时区）
function daysDiff(tsA, tsB) {
  return Math.floor(Math.abs(tsA - tsB) / 86400000)
}

// 本地 MM-DD 字符串（用于生日比对，格式 '03-15'）
function toLocalMMDD(ts) {
  const d = new Date(ts)
  const mm = String(d.getMonth() + 1).padStart(2, '0')
  const dd = String(d.getDate()).padStart(2, '0')
  return `${mm}-${dd}`
}

// 本地日期字符串（用于显示，格式 'YYYY/MM/DD'）
function toLocalDateStr(ts) {
  return new Date(ts).toLocaleDateString('zh-CN')
}

module.exports = { daysDiff, toLocalMMDD, toLocalDateStr }
```

现有代码中 `ts = Date.now()` 已全部正确，无需修改存储逻辑。

**回归验证：** `daysDiff(Date.now(), Date.now() - 4 * 86400000)` 返回 `4`

### P0-4：成本上限保护

**目标：** 防止超长输入崩溃或同角色疯狂循环触发摘要。

**改动文件：** `src/main/services/memory-service.js`

在 `_doRunSummary` 中，准备消息文本后、调用 `generateSummary` 前加入：

```javascript
// ---- 成本保护 ----

// 1. token 粗估（中英文保守估算：每2字符≈1token）
function estimateTokens(text) {
  return Math.ceil(text.length / 2)
}

// 2. 输入截断：超过 6000 tokens 时从最早的消息截断
let inputSlice = slice  // slice 是已准备好的消息数组
let inputText = inputSlice.map(m => `[${m.role}] ${m.content}`).join('\n')
while (estimateTokens(inputText) > 6000 && inputSlice.length > 2) {
  inputSlice = inputSlice.slice(1)
  inputText = inputSlice.map(m => `[${m.role}] ${m.content}`).join('\n')
}

// 3. 每日次数限制：同 sessionId 每日最多 10 次
const todayStr = new Date().toISOString().slice(0, 10)  // UTC YYYY-MM-DD
const dailyKey = `summary_daily_${sessionId}_${todayStr}`
const todayCount = parseInt(getAppStateValue(dailyKey) || '0', 10)
if (todayCount >= 10) {
  return { created: false, reason: 'daily-limit-reached' }
}
```

摘要成功后写入 `summary_history` 并更新计数：

```javascript
// 写入存档
addSummaryHistory({
  sessionId,
  summaryText: summary,
  fromMsgId,
  toMsgId,
  inputTokens: estimateTokens(inputText),
  outputTokens: estimateTokens(summary),
  createdAt: Date.now()
})

// 更新每日计数
setAppStateValue(dailyKey, String(todayCount + 1))
```

```javascript
async _extractAndMergeProfile(conversationText) {
  const extracted = await this.llmService.generateProfileExtraction(conversationText)
  if (!extracted || typeof extracted !== 'object') return

  const existing = this.db.getUserProfile()  // 返回 { name, occupation, traits, notes, ... }

  const toSave = {}

  // 只写入用户未填写的字段（非空才覆盖）
  if (extracted.name && !existing.name) toSave.name = extracted.name
  if (extracted.occupation && !existing.occupation) toSave.occupation = extracted.occupation
  if (extracted.birthday && !existing.birthday) toSave.birthday = extracted.birthday
  if (extracted.birthday_year && !existing.birthday_year) {
    toSave.birthday_year = String(extracted.birthday_year)
  }

  // traits：追加新标签，不覆盖已有
  if (Array.isArray(extracted.traits) && extracted.traits.length > 0) {
    const existingList = (existing.traits || '').split(',').map(s => s.trim()).filter(Boolean)
    const newOnes = extracted.traits.filter(t => t && !existingList.includes(t))
    if (newOnes.length > 0) {
      toSave.traits = [...existingList, ...newOnes].join(', ')
    }
  }

  // notes：追加而非覆盖
  if (extracted.notes_append) {
    const prev = (existing.notes || '').trim()
    toSave.notes = prev ? `${prev}\n---\n${extracted.notes_append}` : extracted.notes_append
  }

  for (const [key, value] of Object.entries(toSave)) {
    this.db.setUserProfileField(key, String(value))
  }
}
```

在 `db.js` 新增 `setUserProfileField(key, value)` 函数（单字段 UPSERT）：

```javascript
function setUserProfileField(key, value) {
  db.prepare(`
    INSERT INTO user_profile (key, value, updated_at) VALUES (?, ?, ?)
    ON CONFLICT(key) DO UPDATE SET value = excluded.value, updated_at = excluded.updated_at
  `).run(key, value, Date.now())
}
```

- 用户已手动填写姓名：不被覆盖

function getCharacterMemory(sessionId) {
  const row = db.prepare('SELECT * FROM character_memory WHERE session_id = ?').get(sessionId)
  return row ? {
    sessionId: row.session_id,
    relationshipStage: row.relationship_stage,
    summaryCount: row.summary_count,
    updatedAt: row.updated_at
  } : { sessionId, relationshipStage: 'new', summaryCount: 0, updatedAt: 0 }
}

function upsertCharacterMemory(sessionId, data) {
  db.prepare(`
    INSERT INTO character_memory (session_id, relationship_stage, summary_count, updated_at)
    VALUES (?, ?, ?, ?)
    ON CONFLICT(session_id) DO UPDATE SET
      relationship_stage = excluded.relationship_stage,
      summary_count = excluded.summary_count,
      updated_at = excluded.updated_at
  `).run(sessionId, data.relationshipStage, data.summaryCount, Date.now())
}

### P2-2：基础主动发话

**目标：** 仅实现两种触发：缺席 > 3 天、生日当天。前端不硬编码话术，由角色 system prompt + LLM 生成。

**触发条件补充说明：**
- 生日字段来自 `user_profile.birthday`（格式 `MM-DD`，由 P1-2 自动提取或用户手动填写）
- `last_open_at` 在每次 app 启动时写入（`app.on('ready')` 时）

**改动文件：** `src/main/main.js`（新增 `checkProactiveGreeting`）

```javascript
const { daysDiff, toLocalMMDD } = require('./utils/time')

async function checkProactiveGreeting() {
  if (!db.getBooleanAppStateValue('proactive_enabled', false)) return

  const now = Date.now()
  const todayUTC = new Date(now).toISOString().slice(0, 10)  // YYYY-MM-DD

  // ---- 缺席触发 ----
  const lastOpenAt = parseInt(db.getAppStateValue('last_open_at') || '0', 10)
  const absent = lastOpenAt ? daysDiff(now, lastOpenAt) : 0

  const absenceLockKey = `proactive_absence_${todayUTC}`
  if (absent > 3 && !db.getAppStateValue(absenceLockKey)) {
    db.setAppStateValue(absenceLockKey, '1')
    setTimeout(() => {
      sendToAllWindows(IPC.PROACTIVE_TRIGGER, { type: 'absence', daysSince: absent })
    }, 2000)
    // 缺席和生日同天不叠加，直接返回
    db.setAppStateValue('last_open_at', String(now))
    return
  }

  // ---- 生日触发 ----
  const birthday = db.getUserProfileField('birthday')  // 格式 'MM-DD'
  if (birthday) {
    const todayMMDD = toLocalMMDD(now)  // 用本地时区比对
    const birthdayLockKey = `proactive_birthday_${todayUTC}`
    if (todayMMDD === birthday && !db.getAppStateValue(birthdayLockKey)) {
      db.setAppStateValue(birthdayLockKey, '1')
      setTimeout(() => {
        sendToAllWindows(IPC.PROACTIVE_TRIGGER, { type: 'birthday' })
      }, 2000)
    }
  }

  db.setAppStateValue('last_open_at', String(now))
}
```

#### db.js 新增

```javascript
function getMemorySummariesForUI(sessionId) {
  return db.prepare(`
    SELECT id, summary, from_msg_id as fromMsgId, to_msg_id as toMsgId, ts
    FROM memory_summaries
    WHERE session_id = ? AND deleted_at IS NULL
    ORDER BY ts DESC
  `).all(sessionId)
}

function softDeleteSummary(id) {
  db.prepare('UPDATE memory_summaries SET deleted_at = ? WHERE id = ?').run(Date.now(), id)
}
```

**注意：** `getRecentMemorySummaries` 也需加 `AND deleted_at IS NULL` 过滤，确保被删摘要不再注入。
